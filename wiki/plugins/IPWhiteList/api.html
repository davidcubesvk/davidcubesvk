How to use:

  • Add IPWhiteList-0.0.jar into project sources.
  • Add depend or softdepend for the IPWhiteList plugin into your plugin's plugin.yml:

    Adding depend or softdepend to your plugin.yml

    depend: [IPWhiteList]
    OR
    softdepend: [IPWhiteList]
  • Get the API class:

    Getting the API class

    IPWhiteListAPI.getInstance();

API classes (2):

com.davidcubesvk.ipWhiteList.api.IPWhiteListAPI.java

package com.davidcubesvk.ipWhiteList.api;

/**
 * API class, includes many methods for basic administration.
 * 
 * Please note, that this class is also used by the plugin itself.
 */
public class IPWhiteListAPI {

    /**
     * Reloads the plugin configuration without kicking all players.
     */
    public void reload();

    /**
     * Reloads the whitelisted IPs.
     */
    public void reloadWhitelistedIPs();

    /**
     * Checks the connecting player's VirtualHost if the IP player used to connect is one of whitelisted IPs.
     * 
     * The virtualHost parameter should be an instance got from {@link PendingConnection#getVirtualHost()} or similar method.
     *
     * @param virtualHost the player's virtual host
     * @return if the IP player used to connect is whitelisted
     */
    public boolean checkConnectingIP(InetSocketAddress virtualHost);

    /**
     * Returns if plugin is in the setup mode.
     *
     * @return if the plugin has setup mode enabled
     */
    public boolean setupMode();

    /**
     * Returns the list containing whitelisted IPs represented by an IPHolder instance.
     *
     * @return the list of whitelisted IPs
     */
    public List<IPHolder> getWhiteListed();

    /**
     * Returns the port, on which is BungeeCord hosted.
     *
     * @return the BungeeCord port
     */
    public int getBungeePort();

    /**
     * Returns the IP of machine on which is BungeeCord hosted.
     *
     * @return the BungeeCord IP
     */
    public String getBungeeIP();

    /**
     * Returns the API instance.
     *
     * @return the API instance
     */
    public static IPWhiteListAPI getInstance();

}

com.davidcubesvk.ipWhiteList.api.IPHolder.java

package com.davidcubesvk.ipWhiteList.api;

/**
 * Class used to store a whitelisted IP and if it should be compared case-sensitively or not.
 */
public class IPHolder {

    /**
     * Initializes the holder with the given values.
     *
     * @param ip            the IP to hold
     * @param caseSensitive if the IP is case-sensitive or not (if should be compared case-sensitively or not)
     */
    public IPHolder(String ip, boolean caseSensitive);

    /**
     * Returns the IP held by this instance of this class.
     *
     * @return the IP held
     */
    public String getIp();

    /**
     * Returns if this IP is case sensitive.
     *
     * @return if this IP is case sensitive
     */
    public boolean isCaseSensitive();

    /**
     * Sets the IP value held by this instance.
     *
     * @param ip the IP to be held
     */
    public void setIp(String ip);

}
