#Setup mode (toggleable using /ipwl setup):
#(set this to false when you set up whitelisted IPs, to make them working. If true, plugin ignores whitelisted IPs)
setup: true
#WhiteListed IPs:
#INFO: 1. To enable all hostnames, insert *. prefix before the whitelisted IP, e.g. *.yourserver.com will allow players connecting with
#         abc.yourserver.com, xyz.yourserver.com..., (useful for servers with AntiDDoS service).
#      2. To apply whitelisted IP without hostnames, insert only the IP without any prefix, e.g. yourserver.com will allow only players
#         connecting with yourserver.com.
#      3. If your server has dynamic IP, use {ip} placeholder as the server's IP (useful for dynamic IP servers).
#      4. To allow not case-sensitive hostname comparing for the current IP, insert "[NOT_CASE_SENSITIVE]" or "[CASE_SENSITIVE]"
#         (for case-sensitive comparing) right before the IP as shown in example whitelisted IPs list. If not inserted, the default
#         setting is case-sensitive.
#NOTE: You don't need to write ports there. IPWhiteList will automatically obtain port where is BungeeCord server hosted on.
#      However, if you write port there, BungeeCord port will be overridden by specified port.
whitelisted:
  - 'yourserver.com'
  - 'localhost'
  - '[NOT_CASE_SENSITIVE]abc.yourserver.com'
#Website for obtaining the {ip} placeholder (any website with only one line displayed - your IP):
ipWebsite: "https://checkip.amazonaws.com/"
#Disconnect messages:
disconnect:
  #If player tries to connect to the server with IP not corresponding with any of whitelisted IPs, disconnect with message (use \n for new line):
  whitelist: "&7[&bIPWhiteList&7] &cPlease connect with IP: yourserver.com!"
  #If SocketServer failed to start (type = SOCKET), or failed to connect to the database server (type = MYSQL) and player tries to join,
  #disconnect with message (use \n for new line):
  failure: "&7[&bIPWhiteList&7] &cWe are now having technical problems.\n\n&cPlease try again in a while."
  #If player tries to connect as player that's already online:
  online: "&7[&bIPWhiteList&7] &cAlready connected to the server!\n\n&cIf you believe this is an error, contact staff."
  #If plugin reloaded, kick all players with message:
  reload: "&7[&bIPWhiteList&7] &cPlugin reloaded.\n\n&cPlease rejoin."
#Logging:
log:
  #Enable logging:
  enabled: true
  #Log the CHECK_CONNECTION data input message (if connect.type = SOCKET):
  #INFO! Connection messages are also logged - e.g. data input and output of socket, what is being written in the database.
  #      If using socket connection, IPWhiteListBukkit sends the CHECK_CONNECTION message to IPWhiteList to check the connection.
  #      Checking connection is set to every second (can change on connect.socket.connectionCheck), so you get this message
  #      every one second in log file. This can lead to 4.3MB/24h logs only from this message.
  logCheckConnection: false
#Connection with extension IPWhiteListBukkit:
connect:
  #Type of the connection (more and detailed info can be found on the wiki):
  #- YML: connection provided by data.yml file created in plugin's data folder
  #- SOCKET: connection using sockets
  #- MYSQL: connection using MySQL database
  type: SOCKET
  #Socket port (if type = SOCKET):
  socketPort: 4444
  #MySQL connection (if type = MYSQL):
  mysql:
    #Host:
    host: ""
    #Port:
    port: ""
    #Username
    username: ""
    #Password:
    password: ""
    #Database:
    database: ""
    #Table name to use:
    table: "ipwhitelist"
    #Use SSL certificate to connect (secure way for communication, please read the official MySQL wiki how to configure SSL communication):
    useSSL: true
    #Delay between reconnecting attempts, when connection has been unexpectedly closed (in seconds):
    reconnect: 3
    #Whitelisted IPs synchronization:
    #INFO! Very useful for large networks with more than one proxy. In this case you don't need to copy whitelisted IPs
    #      anytime you change them to every config, just synchronize whitelisted IPs (all IPWhiteList plugins) with one
    #      central IPWhiteList.
    whitelistedSync:
      #Synchronization mode:
      #- NONE: do not query / upload whitelisted IPs to the database and use the config.yml file for them
      #- HOST: host the whitelisted IPs (central IPWhiteList) - upload whitelisted IPs from config.yml file to the database
      #        so other IPWhiteList plugins set on SYNC mode can query them and synchronize them
      #        WARNING! If changes in whitelisted IPs were made between restarts, central IPWhiteList's server must start as first.
      #- SYNC: synchronize whitelisted IPs with the central IPWhiteList by querying the database
      mode: NONE
      #Table name to store whitelisted IPs in:
      table: "ipwhitelist_ips"
